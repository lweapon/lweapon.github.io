// + 如果存在数字和字符串相加，结果字符串类型，而+会被认为是拼接的意思
var number3 = '50';
var number4 = 50;
var name = '张三';
console.log(number1 + number3);
console.log(number1 + name);
// - 会对数据类型进行“隐式转换”,如果存在数字和'字符串类型的数字'相减，结果是数字类型
//  如果存在数字和数字类型字符串相减，结果是  NaN     not  a  number !
console.log(number1 - number3);  // -20
console.log(number1 - name); // NaN
// 注意： NaN永远 ！= NaN
console.log(NaN == NaN);

console.log(number1/number3);
console.log(number1*number3);
console.log(number1%number3);
console.log('==============')

// 在js中 一般 字符类型的数字会被隐式转换  
// 在比较运算符中会被直接转
console.log(number3 == number4);
//  === 严格等于的意思   1.先比较数据类型  2.再比较数值
// 判断数值相等 === 最靠谱
console.log(number3 === number4);
